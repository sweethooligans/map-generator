  <!-- Google Maps JS API (replace YOUR_API_KEY) -->

  <script src="https://maps.googleapis.com/maps/api/js?key=YOUR_API_KEY&callback=initMap" async defer></script>

  <script>

    // === CONFIG ===

    const MAP_IDS = {

      style1: "e39caf81ba4082b8",

      style2: "7c3917b8837ac562",

      style3: "d9bd015cd976b393"

    };

    const HEART_ICON = "https://upload.wikimedia.org/wikipedia/commons/thumb/f/f1/Heart_coraz%C3%B3n.svg/1920px-Heart_coraz%C3%B3n.svg.png";

    const DEFAULT_CENTER = { lat: 40.7128, lng: -74.0060 }; // NYC fallback

    const DEFAULT_ZOOM = 12;


    let map, marker, geocoder;

    let currentStyleKey = "style1";


    function initMap(){

      geocoder = new google.maps.Geocoder();


      map = new google.maps.Map(document.getElementById("map"), {

        center: DEFAULT_CENTER,

        zoom: DEFAULT_ZOOM,

        mapId: MAP_IDS[currentStyleKey],

        mapTypeControl: false,

        streetViewControl: false,

        fullscreenControl: false,

        clickableIcons: false

      });


      marker = new google.maps.Marker({

        position: DEFAULT_CENTER,

        map,

        draggable: true,

        icon: { url: HEART_ICON, scaledSize: new google.maps.Size(34,34) },

        title: "Drag me to adjust"

      });


      map.addListener("zoom_changed", () => {

        document.getElementById("zoomLevel").textContent = map.getZoom();

      });


      // Search submit

      document.getElementById("address-form").addEventListener("submit", (e)=>{

        e.preventDefault();

        const addr = document.getElementById("address").value.trim();

        if(!addr) return;

        showSpinner(true);

        geocoder.geocode({ address: addr }, (results, status)=>{

          showSpinner(false);

          if(status === "OK" && results[0]){

            const loc = results[0].geometry.location;

            map.panTo(loc);

            marker.setPosition(loc);

            if(map.getZoom() < 14) map.setZoom(14);

          }else{

            alert("Address not found: " + status);

          }

        });

      });


      // Tabs (styles)

      document.querySelectorAll(".tab").forEach(btn=>{

        btn.addEventListener("click", ()=>{

          document.querySelectorAll(".tab").forEach(b=>b.classList.remove("active"));

          btn.classList.add("active");

          currentStyleKey = btn.dataset.style;

          // Swap style by rebuilding map options for reliability

          const center = map.getCenter();

          const zoom = map.getZoom();

          map.setOptions({ mapId: MAP_IDS[currentStyleKey] });

          map.setCenter(center);

          map.setZoom(zoom);

          marker.setMap(map);

        });

      });


      // Zoom controls

      document.getElementById("zoomIn").addEventListener("click", ()=> map.setZoom(map.getZoom()+1));

      document.getElementById("zoomOut").addEventListener("click", ()=> map.setZoom(map.getZoom()-1));

      document.getElementById("centerMarker").addEventListener("click", ()=> map.panTo(marker.getPosition()));


      // Download Static Map

      document.getElementById("downloadBtn").addEventListener("click", downloadStaticPNG);

      // Keyboard UX

      document.getElementById("address").addEventListener("keydown", (e)=>{

        if(e.key === "Enter"){ /* handled by form submit */ }

      });

    }


    function showSpinner(show){

      const s = document.getElementById("spinner");

      s.style.display = show ? "inline-block" : "none";

      s.setAttribute("aria-hidden", show ? "false" : "true");

    }


    function clamp(val, min, max){ return Math.max(min, Math.min(max, val)); }


    function downloadStaticPNG(){

      // Build a Static Maps URL that mirrors the current view.

      const center = map.getCenter();

      const lat = center.lat();

      const lng = center.lng();

      const zoom = clamp(map.getZoom(), 0, 21);

      // Fit within free static size limits: 640x640 with scale=2 => 1280px output

      const viewport = getViewportSize();

      const maxBase = 640;

      // Preserve aspect ratio of the viewport

      let w = Math.min(maxBase, Math.round(viewport.w));

      let h = Math.min(maxBase, Math.round(viewport.h));

      // avoid zero or tiny

      w = Math.max(320, w);

      h = Math.max(320, h);


      const markerPos = marker.getPosition();

      const iconUrl = encodeURIComponent(HEART_ICON);

      const markers = `markers=icon:${iconUrl}|${markerPos.lat()},${markerPos.lng()}`;


      // If your project has Static Maps enabled with map styles in Cloud, you can pass map_id

      const params = new URLSearchParams({

        center: `${lat},${lng}`,

        zoom: String(zoom),

        size: `${w}x${h}`,

        scale: "2",

        format: "png",

        map_id: MAP_IDS[currentStyleKey],

        key: "YOUR_API_KEY"

      });


      // Compose URL

      const url = `https://maps.googleapis.com/maps/api/staticmap?${params.toString()}&${markers}`;


      // Trigger download (prefer fetch to force "download" filename)

      fetch(url)

        .then(r => r.blob())

        .then(blob => {

          const a = document.createElement("a");

          const styleLabel = currentStyleKey.toUpperCase();

          a.href = URL.createObjectURL(blob);

          a.download = `sweet-hooligans-map-${styleLabel}-${Date.now()}.png`;

          document.body.appendChild(a);

          a.click();

          a.remove();

          URL.revokeObjectURL(a.href);

        })

        .catch(err => {

          console.error(err);

          // If fetch is blocked by CORS, fallback to opening the URL in a new tab

          window.open(url, "_blank");

        });

    }


    function getViewportSize(){

      // Get visible map container size for better aspect matching

      const rect = document.getElementById("map").getBoundingClientRect();

      // Leave some margin so UI isn't cropped when composing

      const margin = 40;

      return {

        w: Math.max(320, rect.width - margin),

        h: Math.max(320, rect.height - margin)

      };

    }
